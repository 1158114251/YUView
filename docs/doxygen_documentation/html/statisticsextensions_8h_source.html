162  {
163  // clamp the value to [min max]
164  if (value > rangeMax) value = (float)rangeMax;
165  if (value < rangeMin) value = (float)rangeMin;

167  float span = (float)(rangeMax-rangeMin),
168  x = (value - (float)rangeMin) / span,
169  r = 1,

171  b = 1,
172  a = 1;
173  float F,N,K;
174  int I;
175 
176  switch (type) {
177  case jetColormap:
178  if ((x >= 3.0/8.0) && (x < 5.0/8.0)) r = (4.0f * x - 3.0f/2.0f); else
179  if ((x >= 5.0/8.0) && (x < 7.0/8.0)) r = 1.0; else
180  if (x >= 7.0/8.0) r = (-4.0f * x + 9.0f/2.0f); else
181  r = 0.0f;
182  if ((x >= 1.0/8.0) && (x < 3.0/8.0)) g = (4.0f * x - 1.0f/2.0f); else
183  if ((x >= 3.0/8.0) && (x < 5.0/8.0)) g = 1.0; else
184  if ((x >= 5.0/8.0) && (x < 7.0/8.0)) g = (-4.0f * x + 7.0f/2.0f); else
185  g = 0.0f;
186  if (x < 1.0/8.0) b = (4.0f * x + 1.0f/2.0f); else
187  if ((x >= 1.0/8.0) && (x < 3.0/8.0)) b = 1.0f; else
188  if ((x >= 3.0/8.0) & (x < 5.0/8.0)) b = (-4.0f * x + 5.0f/2.0f); else
189  b = 0.0f;
190  break;
191  case heatColormap:
192  r = 1;
193  g = 0;
194  b = 0;
195  a = x;
196  break;
197  case hsvColormap: // h = x, s = 1, v = 1
198  if (x >= 1.0) x = 0.0;
199  x = x * 6.0f;
200  I = (int) x; /* should be in the range 0..5 */
201  F = x - I; /* fractional part */
202 
203  N = (1.0f - 1.0f * F);
204  K = (1.0f - 1.0f * (1 - F));
205 
206  if (I == 0) { r = 1; g = K; b = 0; }
207  if (I == 1) { r = N; g = 1.0; b = 0; }
208  if (I == 2) { r = 0; g = 1.0; b = K; }
209  if (I == 3) { r = 0; g = N; b = 1.0; }
210  if (I == 4) { r = K; g = 0; b = 1.0; }
211  if (I == 5) { r = 1.0; g = 0; b = N; }
212  break;
213  case hotColormap:
214  if (x < 2.0/5.0) r = (5.0f/2.0f * x); else r = 1.0f;
215  if ((x >= 2.0/5.0) && (x < 4.0/5.0)) g = (5.0f/2.0f * x - 1); else if (x >= 4.0/5.0) g = 1.0f; else g = 0.0f;
216  if (x >= 4.0/5.0) b = (5.0f*x - 4.0f); else b = 0.0f;
217  break;
218  case coolColormap:
219  r = x;
220  g = 1.0f - x;
221  b = 1.0;
222  break;
223  case springColormap:
224  r = 1.0f;
225  g = x;
226  b = 1.0f - x;
227  break;
228  case summerColormap:
229  r = x;
230  g = 0.5f + x / 2.0f;
231  b = 0.4f;
232  break;
233  case autumnColormap:
234  r = 1.0;
235  g = x;
236  b = 0.0;
237  break;
238  case winterColormap:

242  break;
243  case grayColormap:
244  r = x;
245  g = x;
246  b = x;
247  break;
248  case boneColormap:
249  if (x < 3.0/4.0) r = (7.0f/8.0f * x); else if (x >= 3.0/4.0) r = (11.0f/8.0f * x - 3.0f/8.0f);
250  if (x < 3.0/8.0) g = (7.0f/8.0f * x); else

252  if (x >= 3.0/4.0) g = (7.0f/8.0f * x + 1.0f/8.0f);
253  if (x < 3.0/8.0) b = (29.0f/24.0f * x); else if (x >= 3.0/8.0) b = (7.0f/8.0f * x + 1.0f/8.0f);
254  break;
255  case copperColormap:
256  if (x < 4.0/5.0) r = (5.0f/4.0f * x); else r = 1.0f;
257  g = 4.0f/5.0f * x;
258  b = 1.0f/2.0f * x;
259  break;
260  case pinkColormap:
261  if (x < 3.0/8.0) r = (14.0f/9.0f * x); else
262  if (x >= 3.0/8.0) r = (2.0f/3.0f * x + 1.0f/3.0f);
263  if (x < 3.0/8.0) g = (2.0f/3.0f * x); else
264  if ((x >= 3.0/8.0) && (x < 3.0/4.0)) g = (14.0f/9.0f * x - 1.0f/3.0f); else

267  if (x >= 3.0/4.0) b = (2.0f * x - 1.0f);
268  break;
269  case linesColormap:
270  if (x >= 1.0) x = 0.0f;
271  x = x * 7.0f;
272  I = (int) x;
273 
274  if (I == 0) { r = 0.0; g = 0.0; b = 1.0; }
275  if (I == 1) { r = 0.0; g = 0.5; b = 0.0; }
276  if (I == 2) { r = 1.0; g = 0.0; b = 0.0; }
277  if (I == 3) { r = 0.0; g = 0.75; b = 0.75; }
278  if (I == 4) { r = 0.75; g = 0.0; b = 0.75; }
279  if (I == 5) { r = 0.75; g = 0.75; b = 0.0; }
280  if (I == 6) { r = 0.25; g = 0.25; b = 0.25; }
281  break;
282  case col3_gblr_Colormap:
283  if (x < 0.5) { r = 0.0; g = 1.0 - 2 * x; b = 0.0; }
284  else { r = (x - 0.5) * 2; g = 0.0; b = 0.0; }
285  break;
286  case col3_gwr_Colormap:
287  if (x < 0.5) { r = x * 2; g = 1.0; b = x * 2; }
288  else { r = 1.0; g = (1 - x) * 2; b = (1 - x) * 2; }
289  break;
290  case col3_bblr_Colormap:
291  if (x < 0.5) { r = 0.0; g = 0.0; b = 1.0 - 2 * x; }
292  else { r = (x - 0.5) * 2; g = 0.0; b = 0.0; }
293  break;
294  case col3_bwr_Colormap:
295  if (x < 0.5) { r = x * 2; g = x * 2; b = 1.0; }
296  else { r = 1.0; g = (1 - x) * 2; b = (1 - x) * 2; }
297  break;
298  case col3_bblg_Colormap:
299  if (x < 0.5) { r = 0.0; g = 0.0; b = 1.0 - 2 * x; }
300  else { r = 0.0; g = (x - 0.5) * 2; b = 0.0; }
301  break;
302  case col3_bwg_Colormap:
303  if (x < 0.5) { r = x * 2; g = x * 2; b = 1.0; }
304  else { r = (1 - x) * 2; g = 1.0; b = (1 - x) * 2; }
305  break;
306  }

313  return QColor(retR, retG, retB, retA);
